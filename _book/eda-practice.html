每行以x为均值生成一个随机数, 以下哪个是正确的？

# A
tibble(x = 1:5) %>% 
  mutate(normal_mean = rnorm(1, mean = x))     

# B
tibble(x = 1:5) %>% 
  mutate(normal_mean = rnorm(n(), mean = x))

# C
tibble(x = 1:5) %>% 
  mutate(normal_mean = map_dbl(x, ~rnorm(1, mean = .)))

# D
tibble(x = 1:5) %>% 
  mutate(normal_mean = map_dbl(x, ~rnorm(1), mean = .))

# E
tibble(x = 1:5) %>% 
  rowwise() %>% 
  mutate(normal_mean = rnorm(1, mean = x))

以下代码哪些会给出相同的图形?

tb <- tibble(
     x = rep(c(1,2,3), 2),

)
  1. ggplot(tb, aes(x,y)) + geom_line()

  3. ggplot(tb, aes(x,y,fill=group)) + geom_line()
